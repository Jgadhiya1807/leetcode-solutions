class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        profit=0
        minValue = prices[0]
        for i in range(1,len(prices)):
            if(prices[i]<minValue):
                minValue=prices[i]
            else:
                p=prices[i] - minValue
                if(p>profit):
                    profit = p
        return profit
