class Solution:
    def trap(self, a: List[int]) -> int:
        lhb=a[0]
        rhb=a[-1]
        l=0
        r=len(a)-1
        res=0
        while(l<=r):
            if(lhb<=rhb):
                if(a[l]>lhb):
                    lhb=a[l]
                else:
                    res+=lhb-a[l]
                l+=1
            else:
                if(a[r]>rhb):
                    rhb=a[r]
                else:
                    res+=rhb-a[r]
                r-=1
        return res
